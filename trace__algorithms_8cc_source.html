105 std::vector<Measurement> Trapezoid::TrapFilter(uint16_t *trace, const int length,int write) {
106
107 good_trace = false;
108 std::vector<Measurement> retval;
109 int P;
110 double M;
111 float BL[length];//={0};
112 float CFD[length];//={0};
113 float sD[length];//={0};
114 float sP[length];//={0};
115 float sR[length];//={0};
116 float sTrap[length];//={0};
117 float fD[length];//={0};
118 float fP[length];//={0};
119 float fR[length];//={0};
120 float fTrap[length];//={0};
121
122 int TCP = -1;
123 int ZCP = -1;
124 int cfd_frac = -1;
125
126 int ffTrig =0, cfdTrig=0;
127
128 //Pole-zero correction
129 if (tau==-1) {
130 P=0;M=1;
131 }
132 else {
133 P=1;
134 M=1/(std::exp(1/(tau))-1);
135 }
136
137 double mean = Trapezoid::GetBaseline(trace, length);
138
139 //Trapezoid filter
140 for (int k=0;k<length;k++) {
141 BL[k]=trace[k]-mean;
142
143 sD[k]=BL[k];
144 fD[k]=BL[k];
145
146 if (k-sL>=0) { sD[k] -= BL[k-sL]; }
147 if (k-fL>=0) { fD[k] -= BL[k-fL]; }
148
149 if (k-(sG+sL)>=0) { sD[k] -= BL[k-(sG+sL)]; }
150 if (k-(fG+fL)>=0) { fD[k] -= BL[k-(fG+fL)]; }
151
152 if (k-(sG+2*sL)>=0) { sD[k] += BL[k-(sG+2*sL)]; }
153 if (k-(fG+2*fL)>=0) { fD[k] += BL[k-(fG+2*fL)]; }
154
155 sP[k]= sD[k];
156 fP[k]= fD[k];
157
158 if (k-1>=0) { sP[k] += sP[k-1]; }
159 if (k-1>=0) { fP[k] += fP[k-1]; }
160
161 sR[k] = P*sP[k] + M*sD[k];
162 fR[k] = P*fP[k] + M*fD[k];
163
164 sTrap[k] = sR[k]/(M*sL);
165 fTrap[k] = fR[k]/(M);
166
167 if (k-1>=0) { sTrap[k] += sTrap[k-1]; }
168 if (k-1>=0) { fTrap[k] += fTrap[k-1]; }
169
170 CFD[k] = (1-S/8)*fTrap[k] - fTrap[k-D];
171 //Find fast trigger and CFD values
172 if (fTrap[k-1]<=ffThr && fTrap[k]>ffThr && ffTrig==0) {
173 TCP = k-1;
174 ffTrig=1;
175 }
176 if (CFD[k-1]<=cfdThr && CFD[k]>cfdThr && cfdTrig==0) {
177 cfdTrig=1;
178 }
179 if (CFD[k-1]*CFD[k]<0 && cfdTrig==1 && ffTrig==1 && k>TCP && ZCP==-1) {
180 ZCP=k-1;
181 }
182
183 if (ZCP==-1) {
184 cfd_frac = -1;
185 }
186 else {
187 cfd_frac = (32768*CFD[ZCP]/(CFD[ZCP]-CFD[ZCP+1]));
188 }
189 }
190
191 retval.emplace_back("TraceTCP", TCP);
192 retval.emplace_back("TraceZCP", ZCP);
193 retval.emplace_back("TraceCFD", cfd_frac);
194 retval.emplace_back("TraceEnergy", sTrap[TCP + sL + sG -1]);
195
196 if (TCP > 0) { good_trace = true; }
197
198 if (write == 1){
199
200#if ROOT_COMPILE == 1
201 //Draw trap filter
202 TMultiGraph *mgSlow = new TMultiGraph("SlowTrap","SlowTrap");
203 TMultiGraph *mgFast = new TMultiGraph("FastTrap","FastTrap");
204 TMultiGraph *mgCFD = new TMultiGraph("CFD","CFD");
205
206 TGraph *slowTrap = new TGraph();
207 TGraph *fastTrap = new TGraph();
208 TGraph *cfd = new TGraph();
209
210 slowTrap->SetLineColor(kRed);
211 fastTrap->SetLineColor(kBlue);
212 cfd->SetLineColor(kGreen);
213
214 slowTrap->SetLineWidth(2);
215 fastTrap->SetLineWidth(2);
216 cfd->SetLineWidth(2);
217
218 for(int k =0 ;k<length;k++){
219 slowTrap->SetPoint(k,k,sTrap[k]);
220 fastTrap->SetPoint(k,k,fTrap[k]);
221 cfd->SetPoint(k,k,CFD[k]);
222 }
223
224 mgSlow->Add(slowTrap);
225 mgFast->Add(fastTrap);
226 mgCFD->Add(cfd);
227

229 mgFast->Write();

232 }
