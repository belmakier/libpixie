36 CFD Measurement::ProcessCFD(unsigned int data, int frequency) {
37 CFD retval = {};
38 if (frequency == 100) {
39 int CFDTime = mCFDTime100(data);
40 unsigned int CFDForce = mCFDForce100(data);
41 retval = {CFDTime, CFDForce};
42 return retval;
43 }
44 else if (frequency == 250) {
45 int CFDTime = mCFDTime250(data);
46 unsigned int CFDTrigSource = mCFDTrigSource250(data);
47 unsigned int CFDForce = mCFDForce250(data);
48 retval.CFDForce = CFDForce;
49 if (CFDForce==0) {
50 retval.CFDFraction = CFDTime - 16384*static_cast<int>(CFDTrigSource);
51 }
52 else {
53 retval.CFDFraction = 16384;
54 }
55 }
56 else if (frequency == 500) {
57 int CFDTime = mCFDTime500(data);
58 unsigned int CFDTrigSource = mCFDTrigSource500(data);
59 if (CFDTrigSource == 7) {
60 retval.CFDForce = 1;
61 retval.CFDFraction = 40960*4/5;
62 }
63 else {
64 retval.CFDForce = 0;
65 retval.CFDFraction = (CFDTime + 8192*(static_cast<int>(CFDTrigSource)-1))*4/5;
66 }
67 }
68 else {
69 std::cout << "\n" ANSI_COLOR_RED "Waring! Slot has frequency of neither 100 MHz, 250 MHz, or 500 MHz" ANSI_COLOR_RESET<< std::endl;
70 }
71 return retval;
72 }

128 nTraceMeas = 0;
129
130 if (headerLength == 4) {;}
131 else {
132 //read the rest of the header
133 uint32_t otherWords[headerLength-4];
134 if (fread(&otherWords, (size_t) 4, (size_t) headerLength-4, reader->file) != (size_t) headerLength-4) {
135 return -1;
136 }
137 //Read the rest of the header

151 ESumLeading = mESumLeading(otherWords[1]);

153 baseline = mBaseline(otherWords[3]);
154 for (int i=0; i<8; i++) {
155 QDCSums[i]=mQDCSums(otherWords[4+i]);
156 }
157 }
158 }
159
160
161 //skip or proces the trace if recorded
162 auto channel = reader->definition.GetChannel(crateID, slotID, channelNumber);
163 //no trace, do nothing
164 if ((eventLength - headerLength) == 0) {;}
165 //skip trace if not needed
166 else if (outTrace==NULL && !channel->traces){
167 if (fseek(reader->file, (eventLength-headerLength)*4, SEEK_CUR)) {
168 return -1;
169 }
170 }
171 //Read trace

173 //uint16_t trace[traceLength]; // = {0};

175 if (fread(&(this->trace[0]), (size_t) 2, (size_t) traceLength, reader->file) != (size_t) traceLength) {
176 return -1;
177 }
178 if (outTrace!=NULL){
179 memcpy(outTrace, &trace,traceLength*sizeof(uint16_t));
180 }
181 if (channel->traces) {
182 PIXIE::Trace::Algorithm *tracealg = channel->alg;
183 if (!tracealg || !tracealg->is_loaded) {
184 //no trace algorigthm, should never happen

188 good_trace = tracealg->good_trace;
