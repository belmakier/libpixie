124 if (this->file) {

141 int Reader::read() {
142 if (eventCtr == MAX_EVENTS) {
143 eventCtr = 0;
144 }
145
146 this->end = false;
147
148 Event *event = &(events[eventCtr]); //pointer to event
149 int retval = event->read(this, coincWindow, warnings);
150
151 if (retval == 0) {} //successful read
152 else if (retval == -1) { //end of file
153 this->end = true;
154 return -1;
155 }
156 else if (retval == 1) { //same channel pileup
157 this->sameChanPU += 1;
158 }
159
160 //increment counters
161 this->pileups += event->pileups;
162 this->badcfd += event->badcfd;
163 this->subevents += event->nMeas;
164 this->nEvents += 1;
165 this->outofrange += event->outofrange;
166 if (event->mult < 5) {
167 this->mults[event->mult-1] += 1;
168 }
169
170 ++eventCtr;
171
172 return 0;
173 }//Reader::read
