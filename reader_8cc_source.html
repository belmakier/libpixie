197 if (this->file) {
198 return (-1); //file has already been opened
199 }
200
201 if (!(this->file = fopen(path.c_str(), "rb"))) {
202 return (-1);
203 }
204
205 fseek(this->file, 0L, SEEK_END);
206 fileSize = ftell(this->file);
207 rewind(this->file);
208

212 std::cout << "memory mapping fd = " << fd << std::endl;
213 // memory map things

232 int Reader::loadbuffer() {
233 if (PIXIE_READTYPE != 2) { std::cout << "Reader::loadbuffer() not appropriate for PIXIE_READTYPE = " << PIXIE_READTYPE << std::endl; return -1; }
234
235 pBuff = new unsigned int[fileSize/4];
236 pOff = 0;
237 pOffMax = 0;
238 if (!NSCLDAQ) {
239 std::memcpy(&pBuff[0], &buffer[0], fileSize);
240 pOffMax = fileSize/4;
241 }
242 else {
243 while (true) {
244 if (nsclreader->findNextFragment(this) < 0 ) {
245 return -1;
246 }
247 std::memcpy(&pBuff[pOffMax], &buffer[off], 4*sizeof(unsigned int));
248 unsigned int headerLength = (0x1F000 & pBuff[pOffMax]) >> 12;
249 unsigned int eventLength = (0x7FFE0000 & pBuff[pOffMax]) >> 17;
250 off += 4*4;
251 pOffMax += 4;
252 if (headerLength > 4) {
253 std::memcpy(&pBuff[pOffMax], &buffer[off], (headerLength-4)*4);
254 pOffMax += headerLength - 4;
255 off += (headerLength - 4)*4;
256 }
257 if (eventLength-headerLength != 0) {
258 std::memcpy(&pBuff[pOffMax], &buffer[off], (eventLength-headerLength)*4);
259 pOffMax += eventLength-headerLength;
260 off += (eventLength-headerLength)*4;
261 }
262 if (nsclreader->presort) { nsclreader->rib_size -= sizeof(int)*4; }
263 if (off >= fileSize) { break; }
264 if (pOffMax % 10000 == 0) {
265 std::cout << "\rfs : " << off << " / " << fileSize << " pOff = " << pOffMax << std::flush;
266 }
267 }
268 std::cout << "\rfs : " << off << " / " << fileSize << " pOff = " << pOffMax << std::flush;
269 }
270 return 1;

280 int Reader::read() {
281 if (eventCtr == MAX_EVENTS) {
282 eventCtr = 0;
283 }
284
285 this->end = false;
286
287 Event *event = &(events[eventCtr]); //pointer to event
288 int retval = event->read(this, coincWindow, warnings);
289 if (last_time == 0) { first_time = measurements[event->fMeasurements[0]].eventTime; }
290 last_time = measurements[event->fMeasurements[0]].eventTime;
291
292 if (retval == 0) {} //successful read
293 else if (retval == -1) { //end of file
294 this->end = true;
295 return -1;
296 }
297 else if (retval == 1) { //same channel pileup
298 }
299 else if (retval == 2) { //MAX_MEAS_PER_EVENT exceeded
300 std::cerr << "Severe warning! MAX_MEAS_PER_EVENT exceeded" << std::endl;
301 }
302
303 //increment counters
304 this->pileups += event->pileups;
305 this->badcfd += event->badcfd;
306 this->subevents += event->nMeas;
307 this->nEvents += 1;
308 this->outofrange += event->outofrange;
309 if (event->mult < 5) {
310 this->mults[event->mult-1] += 1;
311 }
312
313 ++eventCtr;
314
315 return 0;
